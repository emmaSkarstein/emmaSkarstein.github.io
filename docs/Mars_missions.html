Mars missions

I used this r-bloggers post and this David Witkowski post to learn how to get the data from Wikipedia, and this is the Wikipedia article I used.

# Scraping
library("rvest")
# Melting
library("data.table")

library("dplyr")
library("ggplot2")
library("ggthemes")
library(kableExtra)
url <- "https://en.wikipedia.org/wiki/List_of_missions_to_Mars"
#kept_columns <- c("Record high °F (°C)","Record low °F (°C)")
webpage <- read_html(url)

# Use CSS selectors to scrape the table
table <- html_nodes(webpage,'table.wikitable')[[1]]

# Converting the table to a data frame
table <- html_table(table, header = TRUE)

mars <- table %>%
  bind_rows() %>%
  as_tibble()

knitr::kable(head(mars), booktabs = TRUE) %>%
  kable_styling(font_size = 10)
